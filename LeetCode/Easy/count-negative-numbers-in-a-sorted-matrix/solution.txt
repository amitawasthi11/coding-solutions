class Solution {
public:
    int countNegatives(vector<vector<int>>& grid) {
        int m = grid.size() -1;
        int n = grid[0].size()-1;
        int cnt = 0;
        for(int i=n;i>=0;i--){
            for(int j = m;j>=0;j--){
                if(grid[i][j] <0)cnt++;
                else break;
            }
        }
        return cnt;
    }
};
