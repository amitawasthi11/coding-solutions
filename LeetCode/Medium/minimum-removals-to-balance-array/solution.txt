class Solution {
public:
    int minRemoval(vector<int>& nums, int k) {
        int n = nums.size();
	int minnum = n;
    int right =0;
	sort(nums.begin(),nums.end());
     if (n <= 1) return 0;
	for(int left =0;left<n;left++){
        if (right < left) right = left;
	    while(right<n && 1LL * nums[left]*k>=nums[right]){
	           right++;
        }
	            minnum = min(minnum,n-(right-left));
	    }
	
    return minnum;
    }
};
