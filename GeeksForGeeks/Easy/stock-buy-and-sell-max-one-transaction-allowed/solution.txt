class Solution {
  public:
    int maxProfit(vector<int> &prices) {
        int min = INT_MAX;
        int ans = 0;
        for(int i=0;i<prices.size();i++){
            if(prices[i] < min)min = prices[i];
            int profit = prices[i] - min;
            ans = max(ans,profit);
        }
        return ans;
        
    }
};
