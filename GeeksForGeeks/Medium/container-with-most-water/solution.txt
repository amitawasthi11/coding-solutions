class Solution {
  public:
    int maxWater(vector<int> &arr) {
        int i = 0;
        int j = arr.size()-1;
        int ans = 0;
        while(i<j){
            if(arr[i]>arr[j]){
            ans = max(ans,(j-i)*arr[j]);
            j--;
            }
            else{
            ans =max(ans,(j-i)*arr[i]);
            i++;
            }
           
        }
        return ans;
    }
};
